clear; clc; close all;

% =========================================================================
% 1. ПАРАМЕТРЫ СИГНАЛА И СЕТКИ (БЕЗ ПРОБЕЛОВ)
% =========================================================================
T = 1.0;                         % Период повторения сигнала
t = linspace(-0.5, 3.5, 3000);   % Временная сетка
width = T / 3;                   % Ширина импульса ровно 1/3 периода для непрерывности

% Анонимная функция для формирования треугольного импульса
triangle = @(t_vec, t_center, w, amp) ...
    amp * max(0, 1 - abs(t_vec - t_center) / (w / 2));

% =========================================================================
% 2. ГЕНЕРАЦИЯ НЕПРЕРЫВНОГО СИГНАЛА
% =========================================================================
s = zeros(size(t));

for i = 1:length(t)
    t_mod = mod(t(i), T);
    
    % Центры импульсов распределены строго равномерно внутри периода T
    p1 = triangle(t_mod, width/2,     width,  1.0); % Первый положительный
    p2 = triangle(t_mod, 1.5*width,   width,  1.0); % Второй положительный
    p3 = triangle(t_mod, 2.5*width,   width, -1.0); % Третий отрицательный
    
    s(i) = p1 + p2 + p3;
end

% =========================================================================
% 3. ВЫЧИСЛЕНИЕ ОБНОВЛЕННОГО АМПЛИТУДНОГО СПЕКТРА
% =========================================================================
f = linspace(0.01, 30, 1500); 

spec_tri = @(freq, t_center, w, amp) ...
    amp * (w / 2) * (sinc(freq * (w / 2))).^2 .* exp(-2j * pi * freq * t_center);

% Комплексный спектр с новыми точными центрами
S_complex = spec_tri(f, width/2,   width,  1.0) + ...
            spec_tri(f, 1.5*width, width,  1.0) + ...
            spec_tri(f, 2.5*width, width, -1.0);

S_amplitude = abs(S_complex);

% =========================================================================
% 4. ПОСТРОЕНИЕ ГРАФИКОВ
% =========================================================================
figure;

% График сигнала s(t)
subplot(2,1,1);
plot(t, s, 'b', 'LineWidth', 1.0);
hold on;
plot([min(t) max(t)], [0 0], 'k', 'LineWidth', 0.8);
hold off;
grid on;
title('Временной сигнал s(t) (График №15 — Непрерывный)');
xlabel('Время t');
ylabel('s(t)');
axis([min(t) max(t) -1.1 1.1]);

% График амплитудного спектра |S(f)|
subplot(2,1,2);
plot(f, S_amplitude, 'r', 'LineWidth', 1.2);
grid on;
title('Амплитудный спектр сигнала S(f) (Огибающая)');
xlabel('Частота f');
ylabel('|S(f)|');
xlim([0 max(f)]);
